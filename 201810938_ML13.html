def GaussianKernel(v1, v2, sigma):
    #res= np.exp((-(np.linalg.norm(v1-v2)**2))/(2*sigma**2))   #  1
    #res= np.exp(-(np.sum([v1**2, v2**2,-2*v1*v2]))/(2*sigma**2))  #  2
    res= np.exp(-(np.sum([np.power(v1,2), np.power(v2,2),-2*v1*v2]))/(2*sigma**2))  #  3 ...
    return res
